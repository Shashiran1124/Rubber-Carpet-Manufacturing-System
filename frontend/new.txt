import React, { useState } from 'react';
import axios from 'axios';

const AddMachine = () => {
    const [machineID, setMachineID] = useState('');
    const [date, setDate] = useState('');
    const [status, setStatus] = useState('');
    const [nextGeneralRepairDate, setNextGeneralRepairDate] = useState('');

    const handleSubmit = async (e) => {
        e.preventDefault();
        
        const machineData = {
            machineID,
            date,
            status,
            nextGeneralRepairDate,
        };

        try {
            const response = await axios.post('http://localhost:8070/machine/add', machineData);
            alert('Machine Added Successfully!');
        } catch (error) {
            console.error('There was an error adding the machine!', error);
            alert('Failed to add machine');
        }
    };

    return (
        <div>
            <h1>Add a New Machine</h1>
            <form onSubmit={handleSubmit}>
                <div>
                    <label>Machine ID:</label>
                    <input 
                        type="text" 
                        value={machineID} 
                        onChange={(e) => setMachineID(e.target.value)} 
                        required 
                    />
                </div>
                <div>
                    <label>Date:</label>
                    <input 
                        type="date" 
                        value={date} 
                        onChange={(e) => setDate(e.target.value)} 
                        required 
                    />
                </div>
                <div>
                    <label>Status:</label>
                    <input 
                        type="text" 
                        value={status} 
                        onChange={(e) => setStatus(e.target.value)} 
                        required 
                    />
                </div>
                <div>
                    <label>Next General Repair Date:</label>
                    <input 
                        type="date" 
                        value={nextGeneralRepairDate} 
                        onChange={(e) => setNextGeneralRepairDate(e.target.value)} 
                        required 
                    />
                </div>
                <button type="submit">Add Machine</button>
            </form>
        </div>
    );
};

export default AddMachine;
